# Upload UI Improvement Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Improve upload fields with drag-and-drop zones, image previews, file info display, and submit spinners.

**Architecture:** All changes are in two files: Rust-generated HTML in `form.rs` and vanilla JS in `base.html`. No backend changes, no new endpoints, no new dependencies. Upload fields get a styled drop zone wrapper with visual drag feedback, image preview for both existing and new files, and file metadata display for new selections. Form submit gets a spinner when files are selected.

**Tech Stack:** Rust (HTML generation), vanilla JS, twind (CSS classes), htmx (afterSwap hook)

```rust
    if let Some(obj) = value.and_then(|v| v.as_object()) {

            format!(

            )

            String::new()
        };

            r#"<div class="mb-2 text-sm text-secondary flex items-center gap-3">
    {thumbnail}
    <div>
      <div>Current: <a href="/uploads/file/{hash}/{filename}" class="text-accent underline" target="_blank">{filename}</a></div>
      <div class="text-muted text-xs">{mime}</div>
    </div>
  </div>
  <input type="hidden" name="{name}.__current" value='{json_val}'>"#
        );
    }

    format!(
        r#"<div class="mb-4">
  <label class="block text-sm font-medium text-secondary mb-1">{label}{req_star}</label>
  {current_html}
  <div class="upload-zone border-2 border-dashed border-border rounded-lg p-6 text-center cursor-pointer hover:border-accent transition-colors relative" data-upload-zone>
```

```javascript
function formatFileSize(bytes) {
  if (bytes < 1024) return bytes + ' B';
  if (bytes < 1024 * 1024) return (bytes / 1024).toFixed(1) + ' KB';
  return (bytes / (1024 * 1024)).toFixed(1) + ' MB';
}

function initUploadZones() {
  document.querySelectorAll('[data-upload-zone]:not(.upload-zone-init)').forEach(function(zone) {
    zone.classList.add('upload-zone-init');
    var input = zone.querySelector('[data-upload-input]');
    var prompt = zone.querySelector('.upload-zone-prompt');
    var info = zone.querySelector('.upload-zone-info');
    var preview = zone.querySelector('.upload-zone-preview');
    if (!input) return;

    zone.addEventListener('dragover', function(e) {
      e.preventDefault();
      zone.classList.add('border-accent', 'bg-accent-soft');
    });
    zone.addEventListener('dragleave', function(e) {
      e.preventDefault();
      zone.classList.remove('border-accent', 'bg-accent-soft');
    });
    zone.addEventListener('drop', function(e) {
      e.preventDefault();
      zone.classList.remove('border-accent', 'bg-accent-soft');
      if (e.dataTransfer && e.dataTransfer.files.length > 0) {
        input.files = e.dataTransfer.files;
        input.dispatchEvent(new Event('change', { bubbles: true }));
      }
    });

    input.addEventListener('change', function() {
      if (!input.files || !input.files[0]) return;
      var file = input.files[0];

      // Show file info
      info.textContent = file.name + ' (' + formatFileSize(file.size) + ', ' + (file.type || 'unknown type') + ')';
      info.classList.remove('hidden');
      prompt.classList.add('hidden');

      // Revoke previous preview URL if any
      var prevImg = preview.querySelector('img');
      if (prevImg && prevImg.dataset.objectUrl) {
        URL.revokeObjectURL(prevImg.dataset.objectUrl);
      }
      // Show image preview using safe DOM methods
      while (preview.firstChild) preview.removeChild(preview.firstChild);
      if (file.type && file.type.startsWith('image/')) {
        var url = URL.createObjectURL(file);
        var img = document.createElement('img');
        img.src = url;
        img.alt = 'Preview';
        img.dataset.objectUrl = url;
        img.className = 'h-24 max-w-xs object-contain rounded border border-border-light';
        img.onload = function() { URL.revokeObjectURL(url); };
        preview.appendChild(img);
        preview.classList.remove('hidden');
      } else {
        preview.classList.add('hidden');
      }
    });
  });
}
```

- [ ] **Step 2: Add initSubmitSpinner function**

Add after `initUploadZones`:

```javascript
function initSubmitSpinner() {
  document.querySelectorAll('form[enctype="multipart/form-data"]:not(.spinner-init)').forEach(function(form) {
    form.classList.add('spinner-init');
    form.addEventListener('submit', function() {
      var hasFiles = false;
      form.querySelectorAll('input[type="file"]').forEach(function(input) {
        if (input.files && input.files.length > 0) hasFiles = true;
      });
      if (hasFiles) {
        var btn = form.querySelector('button[type="submit"]');
        if (btn) {
          btn.disabled = true;
          btn.dataset.originalText = btn.textContent;
          btn.textContent = 'Uploading...';
          btn.classList.add('opacity-75');
        }
      }
    });
  });
}

function resetSubmitSpinners() {
  document.querySelectorAll('button[type="submit"][data-original-text]').forEach(function(btn) {
    btn.disabled = false;
    btn.textContent = btn.dataset.originalText;
    btn.classList.remove('opacity-75');
    delete btn.dataset.originalText;
  });
}
```
